\lstdefinelanguage{nesl}
{
	% list of keywords
	morekeywords={
		def,
		else,
		false,
		function,
		if,
		in,
		let,
		then,
		true,
		T,
		F,
	},
	sensitive=true,   % keywords case-sensitive
	morecomment=[l]{--},  % l-line comment
	morecomment=[s]{\{-}{-\}},  % s-start and end
	otherkeywords={:,\&,\#,\%,++},
	morestring=[b]"
}
